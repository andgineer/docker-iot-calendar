20class Weather: 

21 def __init__(self, settings: Dict[str, Any]) -> None: 

22 """Init. 

23 

24 :param settings: 

25 settings['openweathermap_key_file_name'] - name and path of json file 

26 with "key", like 

27 

28 { 

29 "key": "your key from https://home.openweathermap.org/users/sign_up" 

30 } 

31 """ 

32 self.settings = settings 

33 self.key = self.load_key() 

34 

35 def load_key(self) -> Optional[str]: 

36 """Load the API key from the file specified in the settings. 

37 

38 :return: The API key as a string, or None if the key couldn't be loaded. 

39 """ 

40 if WEATHER_KEY_PARAM in self.settings \ 

41 and os.path.isfile(self.settings[WEATHER_KEY_PARAM]): 

42 with open(self.settings[WEATHER_KEY_PARAM], 'r') as key_file: 

43 return json.loads(key_file.read())['key'] 

44 else: 

45 return None

47 def openweathermap_icons(self, icon_code: str, condition_code: Union[str, int]) -> str: 

48 """Convert OpenWeatherMap icon code and condition code to icon file names. 

49 

50 Mostly we use icon code but for specific conditions use specific icon. 

51 Full codes list: http://openweathermap.org/weather-conditions 

52 """ 

53 icons_map = { 

54 '01': 'skc', # clear 

55 '02': 'few', 

56 '03': 'sct', # scattered 

57 '04': 'ovc', # overcast clouds & broken clouds 

58 '09': 'shra', # shower rain 

59 '10': 'hi_shwrs', # ligt rain 

60 '11': 'tsra', # thunderstorm 

61 '13': 'sn', # snow 

62 '50': 'mist', # mist 

63 } 

64 conditions_map = { 

65 '511': 'fzra', # freezing rain 

66 '741': 'fg', # fog 

67 '761': 'dust', 

68 '906': 'ip', # hail 

69 '615': 'mix',# snow and light rain 

70 '616': 'rasn',# snow and rain 

71 '900': 'nsurtsra', # tornado 

72 '902': 'nsurtsra', # hurricane 

73 '500': 'ra1', # light rain 

74 '501': 'ra1', # moderate rain 

75 '711': 'smoke', 

76 '905': 'wind', 

77 '957': 'wind' 

78 } 

79 return conditions_map.get(str(condition_code), icons_map.get(icon_code[:-1], ''))

82 cache_time_seconds=MIN_API_CALL_DELAY_SECONDS,

87 def get_weather(self, latitude: float, longitude: float, days: int = 1, units: str = 'm') -> Optional[Dict[str, Union[List[float], List[str], List[datetime.datetime]]]]: 

88 """Fetch weather data for the given latitude and longitude. 

89 

90 :param latitude: 

91 :param longitude: 

92 :param days: 

93 :param units: m - metric, e - USA 

94 :return: 

95 {'temp_min': [], 'temp_max': [], 'icon': [], 'day': []} 

96 or None if no key or load error 

97 """ 

98 if not self.key: 

99 print("Exiting because no API key") 

100 return None 

101 

102 if units == 'e': 

103 units_code = 'imperial' 

104 elif units == 'm': 

105 units_code = 'metric' 

106 else: 

107 units_code = '' 

108 

109 weather_response = requests.get( 

110 'http://api.openweathermap.org/data/2.5/forecast', 

111 params={ 

112 'units': units_code, 

113 'lat': latitude, 

114 'lon': longitude, 

115 'appid': self.key 

116 } 

117 ) 

118 print('Got weather from openweathermap.org:', weather_response.text[:100]) 

119 weather = weather_response.json() 

120 

121 if str(weather['cod']) != '200': 

122 if str(weather['cod']) == '401' and weather['message'].startswith('Invalid API key'): 

123 print('#'*5, '''You should get your key from https://home.openweathermap.org/users/sign_up 

124 and place it into {}'''.format(self.settings['openweathermap_key_file_name'])) 

125 else: 

126 print('#'*5, 'ERROR:', weather['message']) 

127 return None 

128 

129 highs = [] 

130 lows = [] 

131 icons = [] 

132 dates = [] 

133 for weather_day in weather['list']: 

134 date = datetime.datetime.strptime(weather_day['dt_txt'], '%Y-%m-%d %H:%M:%S') 

135 date = date.replace(hour=0, minute=0, second=0, microsecond=0) 

136 temp_min = weather_day['main']['temp_min'] 

137 temp_max = weather_day['main']['temp_max'] 

138 icon_code = weather_day['weather'][0]['icon'] 

139 condition_code = weather_day['weather'][0]['id'] 

140 icon = self.openweathermap_icons(icon_code, condition_code) 

141 if dates and dates[-1] == date: 

142 if highs[-1] < temp_max: 

143 highs[-1] = temp_max 

144 if lows[-1] > temp_min: 

145 lows[-1] = temp_min 

146 else: 

147 if len(dates) >= days: 

148 break 

149 lows.append(temp_min) 

150 highs.append(temp_max) 

151 icons.append(icon) 

152 dates.append(date) 

153 

154 return { 

155 'temp_min': lows, 

156 'temp_max': highs, 

157 'icon': icons, 

158 'day': dates 

159 }
