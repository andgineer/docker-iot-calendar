1""" 

2 Loads weather from weather.gov. 

3 Works for USA only (NDFD regions). 

4 

5 Usage: 

6 Weather().get_weather(latitude, longitude) 

7 

8""" 

9 

10import datetime 

11from urllib.request import urlopen 

12from xml.dom import minidom 

13 

14 

15class Weather(object): 

16 def get_weather(self, latitude, longitude, days=1, units="m"): 

17 """ 

18 :param latitude: 

19 :param longitude: 

20 :param days: 

21 :param units: m - metric, e - USA 

22 :return: 

23 {'temp_min': [], 'temp_max': [], 'icon': [], 'day': []} 

24 or None if no key or load error 

25 """ 

26 

27 weather_xml = urlopen( 

28 "http://graphical.weather.gov/xml/SOAP_server/ndfdSOAPclientByDay.php?" 

29 "whichClient=NDFDgenByDay&lat={latitude}&lon={longitude}" 

30 "&format=24+hourly&numDays={days}&Unit={units}".format( 

31 latitude=latitude, longitude=longitude, days=days, units=units 

32 ) 

33 ).read() 

34 dom = minidom.parseString(weather_xml) 

35 error = dom.getElementsByTagName("error") 

36 if error: 

37 print(weather_xml) 

38 return None 

39 

40 # Parse temperatures 

41 xml_temperatures = dom.getElementsByTagName("temperature") 

42 highs = [None] * days 

43 lows = [None] * days 

44 for item in xml_temperatures: 

45 if item.getAttribute("type") == "maximum": 

46 values = item.getElementsByTagName("value") 

47 for i in range(len(values)): 

48 highs[i] = int(values[i].firstChild.nodeValue) 

49 if item.getAttribute("type") == "minimum": 

50 values = item.getElementsByTagName("value") 

51 for i in range(len(values)): 

52 lows[i] = int(values[i].firstChild.nodeValue) 

53 

54 xml_icons = dom.getElementsByTagName("icon-link") 

55 icons = [None] * days 

56 for i in range(len(xml_icons)): 

57 icons[i] = ( 

58 xml_icons[i].firstChild.nodeValue.split("/")[-1].split(".")[0].rstrip("0123456789") 

59 ) 

60 

61 xml_day_one = dom.getElementsByTagName("start-valid-time")[0].firstChild.nodeValue[0:10] 

62 day_one = datetime.datetime.strptime(xml_day_one, "%Y-%m-%d") 

63 

64 return {"temp_min": highs, "temp_max": lows, "icon": icons, "day": [day_one]} 

65
